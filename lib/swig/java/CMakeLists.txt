INCLUDE(${SWIG_USE_FILE})

FILE(TO_NATIVE_PATH "${LIBPKMN_SOURCE_DIR}/lib/swig" LIBPKMN_SWIG_SOURCE_DIR)
SET(LIBPKMN_SWIG_SOURCE_DIR ${LIBPKMN_SWIG_SOURCE_DIR} CACHE FILEPATH "Pokemon-sim SWIG source directory")

SET(LIBPKMN_JAVA_INCLUDE_DIRS
    ${LIBPKMN_SOURCE_DIR}/include
    ${LIBPKMN_SWIG_SOURCE_DIR}
    ${Boost_INCLUDE_DIRS}
)

FOREACH(dir ${LIBPKMN_JAVA_INCLUDE_DIRS})
    LIST(APPEND CMAKE_SWIG_FLAGS "-I${dir}")
ENDFOREACH(dir ${LIBPKMN_JAVA_INCLUDE_DIRS})
SET_SOURCE_FILES_PROPERTIES(libpkmn_java_swig.i PROPERTIES CPLUSPLUS ON)

INCLUDE_DIRECTORIES(${LIBPKMN_JAVA_INCLUDE_DIRS} ${JAVA_INCLUDE_PATH} ${JNI_INCLUDE_DIRS})

SET(JAVA_BUILD_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/)
SET(JAVA_INSTALL_DIRECTORY share/java)
FILE(MAKE_DIRECTORY ${JAVA_BUILD_DIRECTORY})
FILE(MAKE_DIRECTORY ${JAVA_BUILD_DIRECTORY})

SET(CMAKE_SWIG_OUTDIR ${JAVA_BUILD_DIRECTORY}/org/nc/LibPKMN)
SET(CMAKE_SWIG_FLAGS -module LibPKMN -package "org.nc.LibPKMN" ${CMAKE_GLOBAL_SWIG_FLAGS})
SWIG_ADD_MODULE(LibPKMN java libpkmn_java_swig.i)
SWIG_LINK_LIBRARIES(LibPKMN pkmn)
IF(UNIX)
    SET_TARGET_PROPERTIES(${SWIG_MODULE_LibPKMN_REAL_NAME} PROPERTIES PREFIX "lib")
    SET_TARGET_PROPERTIES(${SWIG_MODULE_LibPKMN_REAL_NAME} PROPERTIES SUFFIX ".so")
    SET_TARGET_PROPERTIES(${SWIG_MODULE_LibPKMN_REAL_NAME} PROPERTIES COMPILE_FLAGS "-std=c++0x")
ELSEIF(WIN32)
    SET_TARGET_PROPERTIES(${SWIG_MODULE_LibPKMN_REAL_NAME} PROPERTIES PREFIX "")
    SET_TARGET_PROPERTIES(${SWIG_MODULE_LibPKMN_REAL_NAME} PROPERTIES SUFFIX ".dll")
ENDIF(UNIX)

SET(JAVA_SOURCE_CODE ${CMAKE_CURRENT_BINARY_DIR}/org/nc/LibPKMN/*.java)

ADD_CUSTOM_COMMAND(
    OUTPUT ${JAVA_BUILD_DIRECTORY}/LibPKMN.jar
    COMMENT "Creating LibPKMN.jar"
    COMMAND ${Java_JAVAC_EXECUTABLE} -d ${JAVA_BUILD_DIRECTORY} ${JAVA_SOURCE_CODE}
    COMMAND ${Java_JAR_EXECUTABLE} cf ${CMAKE_CURRENT_BINARY_DIR}/LibPKMN.jar -C ${JAVA_BUILD_DIRECTORY} org
    DEPENDS ${SWIG_MODULE_LibPKMN_REAL_NAME}
)

ADD_CUSTOM_TARGET(jpkmn_jar ALL DEPENDS ${JAVA_BUILD_DIRECTORY}/LibPKMN.jar)

INSTALL(
    FILES ${JAVA_BUILD_DIRECTORY}/LibPKMN.jar
    DESTINATION ${JAVA_INSTALL_DIRECTORY}
    COMPONENT "LibPKMN_Java"
)
